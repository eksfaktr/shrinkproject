#include "DEC_PLYR" // Player class

ACTOR ShrinkGun : Pistol
{
	Weapon.SelectionOrder 20
	Weapon.AmmoType "Clip"
	Weapon.AmmoUse 1
	Tag "Shrink Ray"
	States
	{
	Ready:	
		RAYG A 1 A_WeaponReady
		Loop
	Deselect:
		RAYG A 1 A_Lower
		Loop
	Select:
		RAYG A 1 A_Raise
		Loop
	Fire:
		RAYG A 1
		RAYG B 1 A_FireCustomMissile("BlasterShot",0,1,10,0)
		RAYG C 1
		RAYG D 1 A_ReFire
		Goto Ready
	Flash:
		TNT1 A 2 A_Light2
		TNT1 A 2 A_Light1
		TNT1 A 0 A_Light0
		Stop
	}
}
